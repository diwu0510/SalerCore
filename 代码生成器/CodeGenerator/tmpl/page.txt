        {
            BindData();
        }

        protected void F_List_RowCommand(object sender, System.Web.UI.WebControls.GridViewCommandEventArgs e)
        {
            int id = Int32.Parse(e.CommandArgument.ToString());

            if (e.CommandName == "del")

                context.{{tablename}}.Remove(id);
            }

            BindData();
        }

        {
